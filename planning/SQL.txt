CREATE DATABASE messaging;

CREATE TABLE Person (
	ID int NOT NULL AUTO_INCREMENT,
    fName varchar(255),
    lName varchar(255),
    Email varchar(255),
    PRIMARY KEY (ID)
);

CREATE TABLE account (
	ID int,
    username varchar(255),
    password varchar(255),
    verified boolean,
    FOREIGN KEY (ID) REFERENCES Person(ID)
); 

CREATE TABLE messages (
    SentToID int,
    SentFromID int,
    Message varchar(255),
    Message_Rec boolean,
    Message_Read boolean,
    FOREIGN KEY (SentToID) REFERENCES Person(ID),
    FOREIGN KEY (SentFromID) REFERENCES Person(ID)
); 

CREATE TABLE Friends (
    FriReqID int AUTO_INCREMENT,
	PersonA_ID int,
    PersonB_ID int,
    Verified bit DEFAULT 0,
    PRIMARY KEY (FriReqID),
    FOREIGN KEY (PersonA_ID) REFERENCES Person(ID),
    FOREIGN KEY (PersonB_ID) REFERENCES Person(ID)
);
